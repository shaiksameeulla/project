CREATE TABLE ff_d_vendor_office_map (
VENDOR_OFFICE_MAP_ID int(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
VENDOR_OFFICE_REGION_ID int(11) NOT NULL,
OFFICE_ID int(11) NOT NULL,
DT_TO_BRANCH enum('N','Y') NOT NULL DEFAULT 'N',
KEY FK_ff_d_vendor_office_map1 (VENDOR_OFFICE_REGION_ID),
KEY FK_ff_d_vendor_office_map2 (OFFICE_ID),
CONSTRAINT FK_ff_d_vendor_office_map1 FOREIGN KEY (VENDOR_OFFICE_REGION_ID) REFERENCES ff_d_vendor_region_map(VENDOR_REGION_MAP_ID) ON DELETE NO ACTION ON UPDATE CASCADE,
CONSTRAINT FK_ff_d_vendor_office_map2 FOREIGN KEY (OFFICE_ID) REFERENCES ff_d_office (OFFICE_ID) ON DELETE NO ACTION ON UPDATE CASCADE);
